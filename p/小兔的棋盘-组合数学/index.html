#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cstring>
#include <queue>
typedef long long LL;
using namespace std;
LL C[100];
int main() {
	C[0] = 1;
	C[1] = 1;
	for (int i = 2; i <= 80; i++) {
		LL t = 0;
		for (int j = 0; j < i; j++)
			t += C[j]*C[i - j - 1];
		C[i] = t;
	}
	LL N;
	int cnt = 1;
	while (cin >> N) {
		if (N == -1) break;
		cout << cnt++ << " " << N << " " << C[N]*2 << endl;
	}
	return 0;
}
