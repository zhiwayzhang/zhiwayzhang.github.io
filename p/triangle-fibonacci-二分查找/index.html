#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cstring>
#include <queue>
using namespace std;

bool vis[23];

int main() {
	int T;
	cin >> T;
	for (int i = 1; i <= T; i++) {
		int N, ans = 0;
		cin >> N;
		memset(vis, false, sizeof(vis));
		for (int j = 1; j <= N - 1; j++) 
		if (!vis[j]) {
			for (int k = j + 1; k <= N; k++) {
				if (!vis[k]) {
					for (int l = k + 1; l <= j + k - 1; l++)
					if (!vis[l] && l <= N) vis[l] = true, ans++;
					break;
				}
			}
		}
		printf("Case #%d: %d\n", i, ans);
	}
	return 0;
}
