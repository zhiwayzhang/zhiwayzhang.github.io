def GetColumRes(arr, j):
    n = len(arr[:,0])
    colum = np.zeros(n)
    for i in range(j-1, n):
        colum[i] = (arr[i, j-1] - arr[i-1, j-1])/(arr[i, 0] - arr[i-j+1, 0])
    return colum
