#include<iostream>
#include<cstdio>
#include <algorithm>
#include <cstring>
#include <string>
#include <queue>
#include <cmath>
#include <map>
#include <set>
#define debug cout<<0<<endl
#define ll long long
const int MAXN = 1e5 + 10;
const int MOD = 1e9 + 7;
using namespace std;

int N;

struct point {
    ll x, y;
} a[MAXN];

ll x[MAXN]; 
ll prex[MAXN];
ll y[MAXN];
ll prey[MAXN];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> N;
    for (int i = 1; i <= N; i++) {
        int xx, yy;
        cin >> xx >> yy;
        x[i] = a[i].x = (xx + yy);
        y[i] = a[i].y = (xx - yy);    
    }
    sort(x+1, x+N+1);    
    sort(y+1, y+N+1);
    for (int i = 1; i <= N; i++) {
        prex[i] = prex[i-1] + x[i];
        prey[i] = prey[i-1] + y[i];
    }
    ll ans = __LONG_LONG_MAX__;
    for (int i = 1; i <= N; i++) {
        ll temp;
        int lx = lower_bound(x+1,x+N+1,a[i].x) - x;
        int ly = lower_bound(y+1,y+1+N,a[i].y) - y;
        temp = (prex[N]-2*prex[lx]-a[i].x*(N-2*lx)) 
              +(prey[N]-2*prey[ly]-a[i].y*(N-2*ly));
        // cout << " list = "<< a[i].x << ' ' << a[i].y << ' ' << temp << endl;
        ans = min(ans, temp);
    }
    cout << (ans/2);
    return 0;
}
