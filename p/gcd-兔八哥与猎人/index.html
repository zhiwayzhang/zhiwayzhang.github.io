#include<bits/stdc++.h>
#define mem(a,b) memset(a,b,sizeof(a))
typedef long long ll;
typedef unsigned long long ull;
using namespace std;

inline int gcd(int a, int b) {
	return b == 0 ? a : gcd(b,a%b);
}

int main() {
	//freopen("test.txt", "r", stdin);
	ios::sync_with_stdio(false);
	cin.tie(0);
	int N;
	cin >> N;
	while (N--) {
		int ax, ay, bx, by;
		cin >> ax >> ay;
		cin >> bx >> by;
		int xx = abs(ax - bx);
		int yy = abs(ay - by);
		if (gcd(xx,yy) != 1) {
			cout << "yes" << endl;
		} else {
			cout << "no" << endl;
		}
	}
	return 0;
}
