#include <queue>
#include <algorithm>
#include <iostream>
#include <queue>
#include <cstring>
#include <string>
typedef long long LL;
const int MOD = 1e4;
using namespace std;

struct Matrix{
	LL m[2][2];
	void print(){
		for (int i = 0; i < 2; i++) {
			for (int j = 0; j < 2; j++)
				cout << m[i][j] << ' ';
			cout << endl;	
		}
	
	}
} base, ans;

Matrix times(Matrix a, Matrix b) {
	Matrix ans;
	ans.m[0][1] = ans.m[0][0] = ans.m[1][0] = ans.m[1][1] = 0;
	for (int i = 0; i < 2; i++) {
		for (int j = 0; j < 2; j++) {
			for (int k = 0; k < 2; k++)
				ans.m[i][j] = (ans.m[i][j] + a.m[i][k]*b.m[k][j])%MOD;
		}
	}
	return ans;
}

void Matrixpow(LL x) {
	base.m[0][0] = base.m[1][0] = base.m[0][1] = 1;
	base.m[1][1] = 0;
	ans.m[0][0] = ans.m[1][1] = 1;
	ans.m[0][1] = ans.m[1][0] = 0;
	while (x) {
		if (x&1){
			ans = times(ans, base);
		} 
		x >>= 1;
		base = times(base, base);
	}
}
LL N;
int main(){
	while (cin >> N) {
		if (N == -1) break;
		if (N == 0) {
			cout << "0" << endl;
			continue;
		}
		Matrixpow(N);
		cout << ans.m[0][1] % MOD << endl; 
	}	
	return 0;
}
