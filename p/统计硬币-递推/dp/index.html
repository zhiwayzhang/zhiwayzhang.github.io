	//freopen("test.txt", "r", stdin);

	int N, M;
        int ans;		
        int dp[2000][2000];
        int n[] = {1,2,5};

	int T;
	cin >> T;
	while (T--) {
		cin >> N >> M;
		mem(dp,0);
		dp[0][0] = 1;
		for (int i = 0; i < 3; ++i) {
			for (int j = 1; j <= N; ++j) {
				for (int k = n[i]; k <= M; ++k) {
					dp[j][k] += dp[j-1][k-n[i]];
				}
			}
		}
		cout << dp[N][M] << endl;
	} 
	return 0;
}
