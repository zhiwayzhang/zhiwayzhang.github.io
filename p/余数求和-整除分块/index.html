//https://www.luogu.com.cn/problem/P2261
#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <string>
#include <queue>
#include <cmath>
#include <map>
#include <set>
#define mem(a,b) memset(a,b,sizeof(a))
#define debug cout<<0<<endl
#define ll long long
const int MAXN = 1e2 + 10;
const int MOD = 1e9 + 7;
using namespace std;

ll ans = 0;

void solve(ll N, ll K) {
    ans = N*K;
    for (ll l = 1, r; l <= N; l = r + 1) {
        if (l > K) {
            break;
        }
        r = min(N, K/(K/l));
        ans -= (r - l + 1)*(K/l)*(l + r)/2;
    }
    cout << ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    ll N, K; cin >> N >> K;
    solve(N, K);
    return 0;
}
